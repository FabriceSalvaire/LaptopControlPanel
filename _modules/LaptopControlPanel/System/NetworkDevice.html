[docs]    def soft(self, status):

        if status:
            block_status = 'block'
        else:
            block_status = 'unblock'
        self._logger.info("Call rfkill %s %u for device %s" % (block_status, self._index, self.name))
        subprocess.check_call(['/sbin/rfkill', block_status, str(self._index)])
