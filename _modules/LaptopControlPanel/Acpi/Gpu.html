[docs]    def state():
        
        with open(GpuManager.__bbswitch_file__, 'r') as f:
            line = f.readline()
            if 'ON' in line:
                return True
            elif 'OFF' in line:
                return False
            else:
                raise NameError('Unknown bbswitch state')
